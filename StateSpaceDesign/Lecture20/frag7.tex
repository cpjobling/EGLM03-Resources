\[
{\bf{A}} - {\bf{BK}}  =  \left[ {\begin{array}{*{20}c}
	   {( - a_1  - k_1 )} & {( - a_2  - k_2 )} &  \cdots  & {( - a_n  - k_n )}  \\
	   1 & 0 &  \cdots  &  \vdots   \\
	    \vdots  &  \ddots  &  \cdots  &  \vdots   \\
	   0 & 0 &  \cdots  & 0  \\
	\end{array}} \right]
\]
therefore
\begin{equation}
	\label{eq:5}
	\det(s\mathbf{I}-\mathbf{A}+\mathbf{BK}) = s^n + ( - a_1  - k_1 )s^{n-1} + ( - a_2  - k_2 )s^{n-2} + \cdots +  ( - a_n  - k_n ) = 0.
\end{equation}
